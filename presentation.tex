\section{Our Approach}

\begin{frame}{The 6-Stage Pipeline}
  \vspace{0.5cm}
  \begin{center}
    \Large \bfseries
    \begin{tabular}{ccccccc}
      \colorbox{accent1!40}{1} & $\rightarrow$ & 
      \colorbox{accent2!40}{2} & $\rightarrow$ & 
      \colorbox{accent3!40}{3} & $\rightarrow$ & 
      \colorbox{accent1!40}{4} \\
      Data & & Features & & Behavior IR & & LLM \\[0.5cm]
      $\downarrow$ &  & $\downarrow$ &  & $\downarrow$ &  & $\downarrow$ \\[0.5cm]
      \colorbox{accent2!40}{6} & $\leftarrow$ & 
      \colorbox{accent3!40}{5} & $\leftarrow$ & 
      \text{Validation} & & \\
      Eval & & Signatures & & & &
    \end{tabular}
  \end{center}
  
  \vspace{1cm}
  \small
  \begin{itemize}
    \item[\textcolor{accent1}{►}] \textbf{Stages 1-3}: Extract and normalize malware behavior data
    \item[\textcolor{accent2}{►}] \textbf{Stage 4}: Use Llama to understand behaviors semantically
    \item[\textcolor{accent3}{►}] \textbf{Stages 5-6}: Generate and validate CAPEv2 signatures
  \end{itemize}
\end{frame}

\begin{frame}{Stage 1 \& 2: Data \& Feature Extraction}
  \vspace{0.3cm}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \textbf{\textcolor{accent1}{Sandbox Behavioral Data}}\\[0.2cm]
    {\small
    \begin{itemize}
      \item 100 malware samples
      \item 10 distinct families
      \item $\sim$ 500 KB avg per sample
      \item Deeply nested JSON
      \item Inconsistent field naming
    \end{itemize}
    }
    
    \vspace{0.4cm}
    \textbf{\textcolor{accent1}{Extract 50+ Features}}\\[0.2cm]
    {\small
    \begin{itemize}
      \item Process creation/injection
      \item Registry modifications
      \item File I/O patterns
      \item Network connections
      \item API call sequences
    \end{itemize}
    }
    
    \column{0.45\textwidth}
    \textbf{\textcolor{accent2}{Feature Distribution}}\\[0.3cm]
    {\small
    \begin{tcolorbox}[colback=lightgray!50]
      \textbf{File Operations}: 412 events\\
      \textbf{Persistence (RunKey)}: 156\\
      \textbf{Process Injection}: 284\\
      \textbf{C2 Communication}: 203\\
      \textbf{Defense Evasion}: 165
    \end{tcolorbox}
    }
  \end{columns}
\end{frame}

\begin{frame}{Stage 3: Behavior Intermediate Representation}
  \vspace{0.5cm}
  \begin{tcolorbox}[colback=accent1!10, colframe=accent1, boxrule=1pt]
    \textbf{Key Innovation}: Normalize messy CAPEv2 JSON into clean, structured format
  \end{tcolorbox}
  
  \vspace{0.5cm}
  \begin{columns}
    \column{0.48\textwidth}
    {\footnotesize \textbf{Raw CAPEv2 (Messy)}}
    \begin{tcolorbox}[colback=red!5]
      \texttt{nested[api][calls][0][...]} \\
      {\small Inconsistent structure} \\
      {\small Redundant data} \\
      {\small Hard to parse}
    \end{tcolorbox}
    
    \column{0.48\textwidth}
    {\footnotesize \textbf{Behavior IR (Clean)}}
    \begin{tcolorbox}[colback=green!5]
      \texttt{\{processes: [...], \\
        registry: [...], \\
        files: [...], \\
        network: [...]\}}
    \end{tcolorbox}
  \end{columns}
  
  \vspace{0.5cm}
  \textbf{Benefits:}
  \begin{itemize}
    \item LLM can focus on semantics, not JSON parsing
    \item Reproducible and testable
    \item Foundation for multi-output systems
  \end{itemize}
\end{frame}

\begin{frame}{Stage 4: LLM Semantic Analysis}
  \vspace{0.3cm}
  \begin{tcolorbox}[colback=accent3!10, colframe=accent3, boxrule=2pt]
    {\large \textbf{Using Llama (Open-Source Local LLM)}}\\
    {\small Why? Zero cost, full transparency, no API constraints}
  \end{tcolorbox}
  
  \vspace{0.5cm}
  \begin{columns}
    \column{0.5\textwidth}
    \textbf{\textcolor{accent1}{For Each Sample:}}
    \begin{enumerate}
      \item Convert behaviors to natural language
      \item Identify threat patterns
      \item Generate threat assessment
      \item Cache results for reproducibility
    \end{enumerate}
    
    \column{0.5\textwidth}
    \textbf{\textcolor{accent1}{For Each Family:}}
    \begin{enumerate}
      \item Aggregate patterns
      \item Identify common behaviors
      \item Cross-family analysis
      \item Generalization testing
    \end{enumerate}
  \end{columns}
  
  \vspace{0.5cm}
  {\small \textbf{Performance:}}\par
  {\small \textcolor{accent2}{4-6 seconds per sample} | \textcolor{accent2}{\$0.00 total cost} | \textcolor{accent2}{100\% reproducible}}
\end{frame}

\begin{frame}{Stage 5 \& 6: Signature Generation \& Validation}
  \vspace{0.3cm}
  \begin{tcolorbox}[colback=accent2!10, colframe=accent2, boxrule=2pt]
    {\textbf{Generate Production-Ready CAPEv2 Python Signatures}}
  \end{tcolorbox}
  
  \vspace{0.4cm}
  \small
  \begin{columns}[T]
    \column{0.33\textwidth}
    \centering
    \textbf{\textcolor{accent1}{Generate}}\\[0.3cm]
    ✓ LLM writes Python code\\
    ✓ Maps to CAPEv2 API\\
    ✓ Includes behavior tests
    
    \column{0.33\textwidth}
    \centering
    \textbf{\textcolor{accent2}{Validate}}\\[0.3cm]
    ✓ Syntax check (AST)\\
    ✓ Logic verification\\
    ✓ Test against samples
    
    \column{0.33\textwidth}
    \centering
    \textbf{\textcolor{accent3}{Deploy}}\\[0.3cm]
    ✓ Production-ready\\
    ✓ 100\% pass rate\\
    ✓ Detection framework
  \end{columns}
  
  \vspace{0.8cm}
  \begin{tcolorbox}[colback=accent1!5, colframe=darkblue]
    \textbf{Critical Finding:} Multi-step validation caught 15\% of LLM errors\\
    {\small Automation needs validation — manual review is non-negotiable}
  \end{tcolorbox}
\end{frame}

\section{The Challenges}

\begin{frame}{Real Challenges We Faced}
  \vspace{0.5cm}
  \begin{tcolorbox}[colback=red!5, colframe=red, boxrule=2pt]
    {\large \textbf{\textcolor{red}{Challenge 1: JSON Hell}}}
  \end{tcolorbox}
  {\small No two CAPEv2 JSON samples were identical. Different nesting, field names, data types.}
  \hfill \textcolor{green}{\textbf{Solution:}} Custom parsing with 50+ error handlers\\[0.3cm]
  
  \begin{tcolorbox}[colback=orange!5, colframe=orange, boxrule=2pt]
    {\large \textbf{\textcolor{orange}{Challenge 2: Feature Selection}}}
  \end{tcolorbox}
  {\small Which features matter? 50+ potential features, but combinations matter more than individuals.}
  \hfill \textcolor{green}{\textbf{Solution:}} Context-aware behavioral grouping\\[0.3cm]
  
  \begin{tcolorbox}[colback=blue!5, colframe=blue, boxrule=2pt]
    {\large \textbf{\textcolor{blue}{Challenge 3: LLM Code Quality}}}
  \end{tcolorbox}
  {\small Generated code looked good but had subtle bugs (missing imports, API misuse).}
  \hfill \textcolor{green}{\textbf{Solution:}} Multi-step validation pipeline\\[0.3cm]
  
  \begin{tcolorbox}[colback=purple!5, colframe=purple, boxrule=2pt]
    {\large \textbf{\textcolor{purple}{Challenge 4: Edge Cases}}}
  \end{tcolorbox}
  {\small MITRE ATT\&CK mapping: 68-71\% accuracy on Discovery/Lateral Movement.}
  \hfill \textcolor{green}{\textbf{Solution:}} Manual review layer for low-confidence predictions
\end{frame}

\section{Conclusions}

\begin{frame}{Key Takeaways}
  \vspace{0.5cm}
  \Large
  \begin{enumerate}
    \setlength\itemsep{0.8cm}
    
    \item {\textbf{\textcolor{accent1}{Feasible}}} — LLMs can generate production-quality malware signatures (86\% precision)
    
    \item {\textbf{\textcolor{accent2}{Generalizable}}} — Patterns learned transfer to new families (77\% precision on unseen)
    
    \item {\textbf{\textcolor{accent3}{Practical}}} — Reduces analyst time from hours to minutes; zero cost using open-source LLMs
    
    \item {\textbf{\textcolor{orange}{Validated}}} — Multi-stage validation ensures quality; not fully autonomous, requires review
    
    \item {\textbf{\textcolor{purple}{Impactful}}} — Real-world value for security teams and fast-paced threat landscapes
  \end{enumerate}

\begin{frame}{Lessons Learned}
  \vspace{0.5cm}
  \begin{columns}
    \column{0.5\textwidth}
    {\Large \textbf{\textcolor{green}{✓ What Worked}}}\\[0.4cm]
    {\small
    \begin{itemize}
      \item Aggressive caching (saved 2 days)
      \item Modular notebook pipeline
      \item Multi-stage validation
      \item Behavior IR abstraction
      \item Local LLM (cost-effective)
    \end{itemize}
    }
    
    \column{0.5\textwidth}
    {\Large \textbf{\textcolor{red}{✗ What To Improve}}}\\[0.4cm]
    {\small
    \begin{itemize}
      \item Prototype with smaller dataset first
      \item Document data formats early
      \item Add validation from day 1
      \item Manual review earlier
      \item Lock down LLM prompts
    \end{itemize}
    }
  \end{columns}
  
  \vspace{0.8cm}
  \begin{tcolorbox}[colback=orange!10, colframe=orange, boxrule=2pt]
    \centering \textbf{Timeline Reality:} Planned 2 weeks → Actual 4 weeks\\
    {\small Feature extraction and LLM integration took longer than expected}
  \end{tcolorbox}
\end{frame}

\begin{frame}{Honest Limitations}

  {\textbf{\textcolor{accent1}{What This System CAN Do:}}}
  \begin{itemize}
